---
title: "FreeRTOSConfig.h Reference"
description: "Complete reference for all FreeRTOSConfig.h configuration options that control kernel behavior."
---

`FreeRTOSConfig.h` is your application's kernel configuration file. It must be present in your project's include path. Every application has its own `FreeRTOSConfig.h` — copy one from the nearest demo project as a starting point.

## Scheduler configuration

```c
/* Enable preemptive scheduling (1) or cooperative (0) */
#define configUSE_PREEMPTION                    1

/* Enable time slicing for equal-priority tasks */
#define configUSE_TIME_SLICING                  1

/* CPU clock speed — used by some ports for tick timer setup */
#define configCPU_CLOCK_HZ                      ( 168000000UL )

/* Tick rate in Hz. 1000 = 1ms tick, 100 = 10ms tick */
#define configTICK_RATE_HZ                      ( ( TickType_t ) 1000 )

/* Maximum number of task priorities (0 to N-1) */
#define configMAX_PRIORITIES                    ( 5 )

/* Minimal task stack size in words (not bytes) */
#define configMINIMAL_STACK_SIZE                ( ( unsigned short ) 128 )

/* Maximum task name length for debugging */
#define configMAX_TASK_NAME_LEN                 ( 16 )

/* Use 16-bit tick counter (1) or 32-bit (0) — use 0 for most targets */
#define configUSE_16_BIT_TICKS                  0

/* Idle task yields if another task at idle priority is ready */
#define configIDLE_SHOULD_YIELD                 1
```

## Memory configuration

```c
/* Total FreeRTOS heap in bytes */
#define configTOTAL_HEAP_SIZE                   ( ( size_t ) ( 75 * 1024 ) )

/* Enable static allocation APIs (xTaskCreateStatic, etc.) */
#define configSUPPORT_STATIC_ALLOCATION         1

/* Enable dynamic allocation APIs (xTaskCreate, etc.) */
#define configSUPPORT_DYNAMIC_ALLOCATION        1

/* Stack overflow checking: 0=off, 1=basic, 2=thorough */
#define configCHECK_FOR_STACK_OVERFLOW          2
```

## Hook functions

```c
/* Application hooks — set to 1 and implement the function */
#define configUSE_IDLE_HOOK                     0
#define configUSE_TICK_HOOK                     0
#define configUSE_MALLOC_FAILED_HOOK            1
#define configUSE_DAEMON_TASK_STARTUP_HOOK      0

/* Malloc failed hook — called when pvPortMalloc returns NULL */
void vApplicationMallocFailedHook( void );

/* Stack overflow hook */
void vApplicationStackOverflowHook( TaskHandle_t xTask, char *pcTaskName );
```

## Software timers

```c
/* Enable software timers */
#define configUSE_TIMERS                        1

/* Timer task priority */
#define configTIMER_TASK_PRIORITY               ( configMAX_PRIORITIES - 1 )

/* Timer command queue length */
#define configTIMER_QUEUE_LENGTH                10

/* Timer task stack depth in words */
#define configTIMER_TASK_STACK_DEPTH            ( configMINIMAL_STACK_SIZE * 2 )
```

## Co-routines (legacy)

```c
/* Co-routines are a legacy feature; prefer tasks for new designs */
#define configUSE_CO_ROUTINES                   0
#define configMAX_CO_ROUTINE_PRIORITIES         2
```

## Mutexes and semaphores

```c
/* Enable mutexes */
#define configUSE_MUTEXES                       1

/* Enable recursive mutexes */
#define configUSE_RECURSIVE_MUTEXES             1

/* Enable counting semaphores */
#define configUSE_COUNTING_SEMAPHORES           1
```

## Queue sets and task notification

```c
/* Enable queue sets */
#define configUSE_QUEUE_SETS                    1

/* Number of notification values per task (default 1, increase for more) */
#define configTASK_NOTIFICATION_ARRAY_ENTRIES   3
```

## Run-time stats and debugging

```c
/* Enable task run-time statistics collection */
#define configGENERATE_RUN_TIME_STATS           0
#define portCONFIGURE_TIMER_FOR_RUN_TIME_STATS()
#define portGET_RUN_TIME_COUNTER_VALUE()        0

/* Enable Tracealyzer integration */
#define configUSE_TRACE_FACILITY                1

/* Include additional stats in xTaskGetSystemState() */
#define configUSE_STATS_FORMATTING_FUNCTIONS    1
```

## MPU (Memory Protection Unit)

```c
/* Enable MPU support — requires MPU-specific port */
#define configUSE_MPU_WRAPPERS                  1
```

## Cortex-M interrupt priority

```c
/* Lowest interrupt priority (highest number on Cortex-M) */
#define configLIBRARY_LOWEST_INTERRUPT_PRIORITY         15

/* Highest priority from which FreeRTOS FromISR APIs may be called */
#define configLIBRARY_MAX_SYSCALL_INTERRUPT_PRIORITY    5

/* Derived values — don't change */
#define configKERNEL_INTERRUPT_PRIORITY   \
    ( configLIBRARY_LOWEST_INTERRUPT_PRIORITY << (8 - configPRIO_BITS) )
#define configMAX_SYSCALL_INTERRUPT_PRIORITY \
    ( configLIBRARY_MAX_SYSCALL_INTERRUPT_PRIORITY << (8 - configPRIO_BITS) )
```

## API inclusions

Include only the APIs your application uses to minimize code size:

```c
#define INCLUDE_vTaskPrioritySet            1
#define INCLUDE_uxTaskPriorityGet           1
#define INCLUDE_vTaskDelete                 1
#define INCLUDE_vTaskSuspend                1
#define INCLUDE_xResumeFromISR              1
#define INCLUDE_vTaskDelayUntil             1
#define INCLUDE_vTaskDelay                  1
#define INCLUDE_xTaskGetSchedulerState      1
#define INCLUDE_xTaskGetCurrentTaskHandle   1
#define INCLUDE_uxTaskGetStackHighWaterMark 1
#define INCLUDE_xTaskGetIdleTaskHandle      1
#define INCLUDE_eTaskGetState               1
#define INCLUDE_xEventGroupSetBitFromISR    1
#define INCLUDE_xTimerPendFunctionCall      1
#define INCLUDE_xTaskAbortDelay             1
#define INCLUDE_xTaskGetHandle              1
```

<Tip>
  Start with the `FreeRTOSConfig.h` from the demo project closest to your hardware. It will already have the correct interrupt priority settings and clock configuration for that platform.
</Tip>
