---
title: "coreMQTT"
description: "Lightweight MQTT 3.1.1 client library for FreeRTOS and constrained embedded devices."
---

coreMQTT is a small, transport-agnostic MQTT 3.1.1 client. You provide the network send/receive functions (plaintext TCP or TLS), and the library handles framing, keep-alives, and QoS acknowledgements.

## Source location

```
FreeRTOS-Plus/Source/Application-Protocols/coreMQTT/
```

## Connecting

```c
#include "core_mqtt.h"
#include "transport_plaintext.h"

static MQTTContext_t xMQTT;
static uint8_t ucNetBuf[ 1024 ];

/* Network context and transport params for plaintext TCP */
static NetworkContext_t xNetworkContext;
static PlaintextTransportParams_t xPlaintextParams;

void prvMQTTConnect( void )
{
    TransportInterface_t xTransport = { 0 };
    MQTTFixedBuffer_t xFixedBuf = { ucNetBuf, sizeof( ucNetBuf ) };

    /* Connect transport — timeouts are in milliseconds */
    xNetworkContext.pParams = &xPlaintextParams;
    Plaintext_FreeRTOS_Connect( &xNetworkContext, "broker.example.com", 1883,
                                5000, 5000 );

    /* Wire transport interface to the network context */
    xTransport.pNetworkContext = &xNetworkContext;
    xTransport.send            = Plaintext_FreeRTOS_send;
    xTransport.recv            = Plaintext_FreeRTOS_recv;

    MQTT_Init( &xMQTT, &xTransport, FreeRTOS_GetTickCount,
               prvEventCallback, &xFixedBuf );

    MQTTConnectInfo_t xConn = {
        .cleanSession           = true,
        .pClientIdentifier      = "my-device-001",
        .clientIdentifierLength = 13,
        .keepAliveSeconds       = 60,
    };

    bool sessionPresent;
    MQTT_Connect( &xMQTT, &xConn, NULL, pdMS_TO_TICKS( 5000 ), &sessionPresent );
}
```

## Publishing

```c
MQTTPublishInfo_t xPub = {
    .qos             = MQTTQoS1,
    .pTopicName      = "device/temperature",
    .topicNameLength = 18,
    .pPayload        = "{\"temp\":23.5}",
    .payloadLength   = 13,
};

MQTT_Publish( &xMQTT, &xPub, MQTT_GetPacketId( &xMQTT ) );
```

## Subscribing

```c
MQTTSubscribeInfo_t xSub = {
    .qos               = MQTTQoS1,
    .pTopicFilter      = "device/commands/#",
    .topicFilterLength = 17,
};

MQTT_Subscribe( &xMQTT, &xSub, 1, MQTT_GetPacketId( &xMQTT ) );
```

## Event callback

```c
static void prvEventCallback( MQTTContext_t *pxCtx,
                               MQTTPacketInfo_t *pxPacket,
                               MQTTDeserializedInfo_t *pxInfo )
{
    if( pxPacket->type == MQTT_PACKET_TYPE_PUBLISH )
    {
        MQTTPublishInfo_t *pxPub = pxInfo->pPublishInfo;
        /* Handle incoming message: pxPub->pPayload, pxPub->payloadLength */
    }
}
```

## Process loop

Call `MQTT_ProcessLoop()` regularly to drive keep-alives and receive incoming packets:

```c
static void vMQTTTask( void *pvParameters )
{
    for( ;; )
    {
        MQTT_ProcessLoop( &xMQTT );
        vTaskDelay( pdMS_TO_TICKS( 10 ) );
    }
}
```

## TLS transport (AWS IoT Core)

```c
#include "transport_mbedtls.h"

static NetworkContext_t xNetworkContext;
static TlsTransportParams_t xTlsParams;

NetworkCredentials_t xCreds = {
    .pRootCa        = (const uint8_t *) pcRootCA,
    .rootCaSize     = sizeof( pcRootCA ),
    .pClientCert    = (const uint8_t *) pcDeviceCert,
    .clientCertSize = sizeof( pcDeviceCert ),
    .pPrivateKey    = (const uint8_t *) pcPrivKey,
    .privateKeySize = sizeof( pcPrivKey ),
};

xNetworkContext.pParams = &xTlsParams;
/* Timeouts are in milliseconds */
TLS_FreeRTOS_Connect( &xNetworkContext, pcAWSEndpoint, 8883,
                      &xCreds, 5000, 5000 );

xTransport.pNetworkContext = &xNetworkContext;
xTransport.send            = TLS_FreeRTOS_send;
xTransport.recv            = TLS_FreeRTOS_recv;
```

<Note>
  For multi-task MQTT sharing, use the **coreMQTT-Agent** library (`FreeRTOS-Plus/Source/Application-Protocols/coreMQTT-Agent/`). It serializes access to a single MQTT connection across multiple application tasks.
</Note>
